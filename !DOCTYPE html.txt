let curFilter="all", curSearch="";

function renderAll(){
  const wrap=document.getElementById("catalogWrap");
  wrap.innerHTML="";
  let anyVisible=false;
  SECTIONS.forEach(sec=>{
    const prods=PRODUCTS.filter(p=>{
      const catOk=curFilter==="all"||p.cat===sec.key;
      const srchOk=curSearch===""||p.name.toLowerCase().includes(curSearch)||p.short.toLowerCase().includes(curSearch);
      return p.cat===sec.key&&catOk&&srchOk;
    });
    if(!prods.length)return;
    anyVisible=true;
    const secEl=document.createElement("div");
    secEl.innerHTML=`
      <div class="sec-head">
        <div class="sec-icon"><i class="${sec.icon}"></i></div>
        <div class="sec-info"><div class="sec-name">${sec.name}</div><div class="sec-sub">${sec.sub}</div></div>
        <div class="sec-count">${prods.length}</div>
      </div>
      <div class="products-grid" id="g-${sec.key}"></div>
    `;
    wrap.appendChild(secEl);
    const grid=secEl.querySelector(`#g-${sec.key}`);
    prods.forEach(p=>{
      const card=document.createElement("div");
      card.className="prod-card";
      card.onclick=()=>openModal(p);
      card.innerHTML=`
        <div class="card-gallery" id="cg-${p.id}">
          ${p.imgs.map((img,i)=>`<img src="${img}" alt="${p.name}" class="${i===0?'active':''}" loading="lazy"/>`).join("")}
          <div class="gallery-dots">
            ${p.imgs.map((_,i)=>`<div class="gallery-dot ${i===0?'active':''}" onclick="event.stopPropagation();cardDot(${p.id},${i})"></div>`).join("")}
          </div>
          <div class="card-badge">${sec.name.split(" ")[0]}</div>
        </div>
        <div class="card-body">
          <div class="card-cat">${sec.name}</div>
          <div class="card-name">${p.name}</div>
          <div class="card-short">${p.short}</div>
          <div class="card-prices">
            <div>
              <div class="price-new-sm">${cop(p.pNew)}</div>
              <div class="price-new-lbl">Nuevo</div>
            </div>
            <div>
              <div class="price-used-sm">${cop(p.pUsed)}</div>
              <div class="price-new-lbl">Usado</div>
            </div>
          </div>
        </div>
        <div class="card-footer">
          <button class="btn-wa-sm" onclick="event.stopPropagation();askWA('${p.name}',${p.pNew},${p.pUsed})">
            <i class="fab fa-whatsapp"></i> Preguntar
          </button>
          <button class="btn-expand" onclick="event.stopPropagation();openModal(PRODUCTS.find(x=>x.id===${p.id}))">
            <i class="fas fa-expand-alt"></i>
          </button>
        </div>
      `;
      // Auto rotate gallery
      let idx=0;
      setInterval(()=>{
        idx=(idx+1)%p.imgs.length;
        const g=document.getElementById(`cg-${p.id}`);
        if(!g)return;
        g.querySelectorAll("img").forEach((im,i)=>im.classList.toggle("active",i===idx));
        g.querySelectorAll(".gallery-dot").forEach((d,i)=>d.classList.toggle("active",i===idx));
      },2800+p.id*200);
      grid.appendChild(card);
    });
  });
  if(!anyVisible)wrap.innerHTML=`<div class="no-results"><i class="fas fa-search"></i><p>No se encontraron productos.</p></div>`;
}

function cardDot(pid,idx){
  const g=document.getElementById(`cg-${pid}`);
  if(!g)return;
  g.querySelectorAll("img").forEach((im,i)=>im.classList.toggle("active",i===idx));
  g.querySelectorAll(".gallery-dot").forEach((d,i)=>d.classList.toggle("active",i===idx));
}

function filterCat(cat,btn){
  curFilter=cat;
  document.querySelectorAll(".filt-btn").forEach(b=>b.classList.remove("active"));
  btn.classList.add("active");
  renderAll();
}

function doSearch(){
  curSearch=document.getElementById("searchInput").value.toLowerCase().trim();
  renderAll();
}

// MODAL
function openModal(p){
  const secName=SECTIONS.find(s=>s.key===p.cat)?.name||p.cat;
  document.getElementById("mCat").textContent=secName;
  document.getElementById("mTitle").textContent=p.name;
  document.getElementById("mDesc").textContent=p.desc;
  document.getElementById("mFeats").innerHTML=p.feats.map(f=>`<div class="modal-feat">${f}</div>`).join("");
  document.getElementById("mPriceNew").textContent=cop(p.pNew);
  document.getElementById("mPriceUsed").textContent=cop(p.pUsed);
  // gallery
  modalImgs=p.imgs; modalImgIdx=0;
  buildModalGallery();
  document.getElementById("mWaBtn").onclick=()=>askWA(p.name,p.pNew,p.pUsed);
  document.getElementById("modalOverlay").classList.add("open");
  document.body.style.overflow="hidden";
}

function buildModalGallery(){
  const g=document.getElementById("modalGallery");
  g.querySelectorAll("img").forEach(im=>im.remove());
  modalImgs.forEach((src,i)=>{
    const im=document.createElement("img");
    im.src=src; im.alt="producto";
    if(i===modalImgIdx)im.classList.add("active");
    g.insertBefore(im,g.firstChild);
  });
  const nav=document.getElementById("modalGallNav");
  nav.innerHTML=modalImgs.map((_,i)=>`<button class="modal-gall-dot ${i===modalImgIdx?'active':''}" onclick="goModalImg(${i})"></button>`).join("");
}

function goModalImg(i){
  modalImgIdx=i;
  buildModalGallery();
}
function modalPrev(){goModalImg((modalImgIdx-1+modalImgs.length)%modalImgs.length)}
function modalNext(){goModalImg((modalImgIdx+1)%modalImgs.length)}

function closeModal(){
  document.getElementById("modalOverlay").classList.remove("open");
  document.body.style.overflow="";
}
function closeModalBg(e){if(e.target===document.getElementById("modalOverlay"))closeModal();}

function askWA(name,pNew,pUsed){
  const msg=`Hola! Me interesa: ${name} (Nuevo: ${cop(pNew)} / Usado: ${cop(pUsed)}). Podrias darme mas informacion?`;
  window.open(`https://wa.me/${WA}?text=${encodeURIComponent(msg)}`,"_blank");
}
